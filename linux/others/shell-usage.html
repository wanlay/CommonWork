# awk

TIP

NR,表示awk开始执行程序后所读取的数据行数
FNR,与NR功用类似,不同的是awk每打开一个新文件,FNR便从0重新累计

输出netstat -n结果的第2 、6、1列

netstat -n | awk '{print $2,$6,$1}'
# 把第2、3列的合输出在最后列
netstat -n | awk '{print $0,$2+$3}'

输出netstat -n结果中，筛选出第二列等于2的

netstat -n | awk '$2 == 2'
ps -ef | awk '$1 == "ubuntu"'
#不等于
# netstat -n | awk '$2 != "2"'

输出netstat -n结果中的第一列是否满足正则表达式，0不满足，1满足

netstat -n | awk '{ print $1 ~ /^[a-p]/}'
# awk '$7  ~ /^[a-f]/' file.txt
# 不匹配
# netstat -n | awk '{ print $1 ！~ /^[a-p]/}'

统计｀netstat -n｀的第二列中各结果的出现次数

netstat -n | awk '{arr[$2]++} END {for(i in arr)print i, arr[i]}'

在netstat -n的结果中，筛选出第一次出现第二列值的结果

netstat -n | awk '!arr[$2]++'

筛选出file.txt中，第3项大于第5项的结果

awk '$3>$5' file.txt

累加ps结果的第一项

ps | awk '{sum+=$1} END {print sum}'

计算netstat -n第二列结果的平均值

netstat -n | awk '{x+=$2}END{print x/NR}'

输出ps结果的除第一行的所有行

ps | awk 'NR>1'
# 输出20-80行
netstat -n | awk 'NR>=20&&NR<=80'

# sed

sed 将ps的输出结果中CMD全部替换为TMD

ps | sed 's/CMD/TMD/g'
# 替换第二个匹配项
ps | sed 's/CMD/TMD/2'
# 替换前三个匹配项
ps | sed 's/CMD/TMD/3g'
# 打印匹配项
ps | sed 's/CMD/TMD/p'

去除ps结果中的空格和\t

# 去除开头的
ps | sed 's/^[ \t]*//'
# 去除结尾的
ps | sed 's/[ \t]*$//'
# 开头和结尾
ps | sed 's/^[ \t]*//;s/[ \t]*$//'

# functions

解压函数

extract () {
   if [ -f $1 ] ; then
       case $1 in
        *.tar.bz2)      tar xvjf $1 ;;
        *.tar.gz)       tar xvzf $1 ;;
        *.tar.xz)       tar Jxvf $1 ;;
        *.bz2)          bunzip2 $1 ;;
        *.rar)          unrar x $1 ;;
        *.gz)           gunzip $1 ;;
        *.tar)          tar xvf $1 ;;
        *.tbz2)         tar xvjf $1 ;;
        *.tgz)          tar xvzf $1 ;;
        *.zip)          unzip $1 ;;
        *.Z)            uncompress $1 ;;
        *.7z)           7z x $1 ;;
        *)              echo "don't know how to extract '$1'..." ;;
       esac
   else
       echo "'$1' is not a valid file!"
   fi
}
